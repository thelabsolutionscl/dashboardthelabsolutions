Conexión Google Ads
⚙ Configuración — 2 scripts obligatorios
Por qué 2 scripts: AdsApp solo existe dentro de Google Ads y no puede desplegarse como App Web. Se necesita un puente.

Script 1 — Apps Script (script.google.com) = el endpoint web:
Crea un proyecto nuevo → pega el código → Desplegar → App Web → Ejecutar como: Yo → Acceso: Cualquier persona → copia la URL al campo Endpoint arriba.

Script 2 — Google Ads Script (ads.google.com → Herramientas → Scripts):
Crea un nuevo script → pega el 2° bloque → actualiza ENDPOINT con la URL del Script 1 → ejecútalo manualmente una vez y programa diario.
Script 1 — Apps Script (script.google.com) — el endpoint web
// ── Script 1: Apps Script (script.google.com) ──────────────────────────
// Proyecto nuevo → pega este código → Desplegar → App Web
// Ejecutar como: Yo  |  Quién tiene acceso: Cualquier persona

var SECRET = 'thelab2025';

function doPost(e) {
  try {
    var body = JSON.parse(e.postData.contents);
    if (body.secret !== SECRET) return json({ok:false, error:'No autorizado'});

    if (body.type === 'mutation') {
      var muts = JSON.parse(PropertiesService.getScriptProperties().getProperty('ads_mutations') || '[]');
      muts.push({op:body.op, id:body.id||'', data:body.data, timestamp:body.timestamp||new Date().toISOString(), status:'pending'});
      PropertiesService.getScriptProperties().setProperty('ads_mutations', JSON.stringify(muts));
      return json({ok:true, queued:true, pending:muts.filter(function(m){return m.status==='pending';}).length});
    }

    if (body.type === 'update_mutations') {
      PropertiesService.getScriptProperties().setProperty('ads_mutations', JSON.stringify(body.mutations));
      return json({ok:true});
    }

    PropertiesService.getScriptProperties().setProperty('ads_cache', JSON.stringify({
      guardado: new Date().toISOString(),
      gasto: body.gasto, impresiones: body.impresiones, clics: body.clics,
      conversiones: body.conversiones, valor_conversion: body.valor_conversion,
      campanas: body.campanas
    }));
    return json({ok:true, saved:true});
  } catch(err) { return json({ok:false, error:err.message}); }
}

function doGet(e) {
  try {
    var action = e && e.parameter && e.parameter.action;
    if (action === 'mutations') {
      var muts = JSON.parse(PropertiesService.getScriptProperties().getProperty('ads_mutations') || '[]');
      return json({ok:true, mutations:muts});
    }
    var days = parseInt((e && e.parameter && e.parameter.days) || 30);
    var raw = PropertiesService.getScriptProperties().getProperty('ads_cache');
    if (!raw) return json({ok:false, error:'Sin datos aún — ejecuta el Script 2 en Google Ads primero'});
    var data = JSON.parse(raw);
    data.ok = true;
    data.periodo = days + ' días';
    return json(data);
  } catch(err) { return json({ok:false, error:err.message}); }
}

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
Script 2 — Google Ads Script (ads.google.com → Herramientas → Scripts)
// ── Script 2: Google Ads Script ────────────────────────────────────────
// ads.google.com → Herramientas → Scripts → + Nuevo script
// Actualiza ENDPOINT con la URL del Script 1. Programa ejecución: diaria.

var ENDPOINT = 'PEGA_AQUI_LA_URL_DEL_SCRIPT_1';
var SECRET   = 'thelab2025';

function main() {
  // 1. Sincronizar datos al Script 1
  var report = AdsApp.report(
    'SELECT campaign.name, campaign.id, campaign.status, ' +
    'campaign_budget.amount_micros, metrics.impressions, metrics.clicks, ' +
    'metrics.cost_micros, metrics.conversions, metrics.conversions_value ' +
    'FROM campaign WHERE segments.date DURING LAST_30_DAYS ' +
    'ORDER BY metrics.cost_micros DESC'
  );
  var rows = report.rows();
  var campaigns = [], t = {gasto:0,impresiones:0,clics:0,conversiones:0,valor_conversion:0};
  while (rows.hasNext()) {
    var row = rows.next();
    var g  = Math.round(parseInt(row['metrics.cost_micros'])/1e6);
    var i  = parseInt(row['metrics.impressions']);
    var c  = parseInt(row['metrics.clicks']);
    var cv = parseFloat(row['metrics.conversions']);
    var vc = parseFloat(row['metrics.conversions_value']);
    t.gasto+=g; t.impresiones+=i; t.clics+=c; t.conversiones+=cv; t.valor_conversion+=vc;
    campaigns.push({
      id:row['campaign.id'], nombre:row['campaign.name'], estado:row['campaign.status'],
      presupuesto:Math.round(parseInt(row['campaign_budget.amount_micros']||0)/1e6),
      gasto:g, impresiones:i, clics:c, conversiones:cv, valor_conversion:vc
    });
  }
  UrlFetchApp.fetch(ENDPOINT, {
    method:'post', contentType:'application/json', muteHttpExceptions:true,
    payload:JSON.stringify({secret:SECRET, campanas:campaigns,
      gasto:t.gasto, impresiones:t.impresiones, clics:t.clics,
      conversiones:t.conversiones, valor_conversion:t.valor_conversion})
  });

  // 2. Procesar mutaciones pendientes del dashboard
  try {
    var mutResp = UrlFetchApp.fetch(ENDPOINT+'?action=mutations', {muteHttpExceptions:true});
    var mutData = JSON.parse(mutResp.getContentText());
    if (!mutData.ok || !mutData.mutations) return;
    var muts = mutData.mutations;
    var changed = false;
    for (var k=0; k<muts.length; k++) {
      var m = muts[k];
      if (m.status !== 'pending') continue;
      try {
        if (m.op === 'create') {
          var budget = AdsApp.newCampaignBudgetBuilder()
            .withName('Budget_'+m.data.nombre)
            .withAmount(m.data.presupuesto/1000)
            .build().getResult();
          AdsApp.newCampaignBuilder()
            .withName(m.data.nombre)
            .withBudget(budget)
            .withStatus(m.data.estado||'ENABLED')
            .withSearchSettings().campaigns().build();
          muts[k].status = 'aplicado';
        } else if (m.op === 'edit') {
          var iter = AdsApp.campaigns().withIds([m.id]).get();
          if (iter.hasNext()) {
            var camp = iter.next();
            if (m.data.nombre) camp.setName(m.data.nombre);
            if (m.data.estado === 'ENABLED') camp.enable();
            if (m.data.estado === 'PAUSED') camp.pause();
            if (m.data.presupuesto) camp.getBudget().setAmount(m.data.presupuesto/1000);
            muts[k].status = 'aplicado';
          }
        } else if (m.op === 'delete') {
          var iter = AdsApp.campaigns().withIds([m.id]).get();
          if (iter.hasNext()) { iter.next().pause(); muts[k].status = 'aplicado'; }
        }
        changed = true;
      } catch(e) {
        muts[k].status = 'error';
        muts[k].error = e.message;
        changed = true;
      }
    }
    if (changed) {
      UrlFetchApp.fetch(ENDPOINT, {
        method:'post', contentType:'application/json', muteHttpExceptions:true,
        payload:JSON.stringify({secret:SECRET, type:'update_mutations', mutations:muts})
      });
    }
  } catch(e) { Logger.log('Mutations error: '+e.message); }
}

⚠ Requerido: activar escritura SEO en WordPress

Para que el botón "Guardar" funcione, debes agregar este código a functions.php de tu tema en WordPress (Apariencia → Editor de temas → functions.php):

add_action( 'rest_api_init', function () {
    foreach ( [ 'post', 'page' ] as $post_type ) {
        register_post_meta( $post_type, '_yoast_wpseo_title', [
            'show_in_rest'  => true,
            'single'        => true,
            'type'          => 'string',
            'auth_callback' => function() { return current_user_can( 'edit_posts' ); },
        ] );
        register_post_meta( $post_type, '_yoast_wpseo_metadesc', [
            'show_in_rest'  => true,
            'single'        => true,
            'type'          => 'string',
            'auth_callback' => function() { return current_user_can( 'edit_posts' ); },
        ] );
    }
} );

Después de agregar el código, haz clic en "Verificar guardado SEO" para confirmar que funciona.
